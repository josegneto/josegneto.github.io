void Display(unsigned char dat[8][8])

{

  for(int i = 0; i<150; i++)

  {

    for(int c = 0; c<8; c++)

    {

        digitalWrite(Column[c],LOW);

       

        for(int r = 0; r<8; r++)

        {

            digitalWrite(Row[r],dat[r][c]);

        }

        delay(1);

        Clear(); 

    }

  }

}

 

void Clear()                              

{

    for(int i = 0; i<8; i++)

    {

        digitalWrite(Row[i],LOW);

        digitalWrite(Column[i],HIGH);

    }

}
